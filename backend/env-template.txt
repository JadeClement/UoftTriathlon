# Push Notifications Configuration

# For iOS (APNs) - Required for iOS push notifications
# Create APNs Auth Key in Apple Developer Portal: https://developer.apple.com/account/resources/authkeys/list
# Download the .p8 file and note the Key ID and Team ID

# For LOCAL DEVELOPMENT: Use file path
APNS_KEY_PATH=./config/AuthKey_XXXXXXXXXX.p8

# For PRODUCTION (Vercel): Use base64 encoded key (convert with: base64 -i AuthKey_XXX.p8 | tr -d '\n')
# APNS_KEY_BASE64=LS0tLS1CRUdJTiBQUklWQVRFIEtFWS0tLS0t...

# Required for both local and production:
APNS_KEY_ID=your_apns_key_id  # e.g., 9PKQNJ3X73
APNS_TEAM_ID=your_apple_team_id  # Your Apple Team ID (found in Apple Developer Portal)
APNS_BUNDLE_ID=uofttri.club.app
# CRITICAL: APNs Production vs Development
# Development builds from Xcode generate tokens for the DEVELOPMENT APNs gateway
# Production builds (App Store/TestFlight) generate tokens for the PRODUCTION APNs gateway
# These are NOT interchangeable - using the wrong gateway causes BadDeviceToken errors
# 
# For testing with Xcode builds: APNS_PRODUCTION=false
# For App Store/TestFlight builds: APNS_PRODUCTION=true
APNS_PRODUCTION=false
